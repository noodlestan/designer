import { Card, CardGrid, Code, LinkButton } from '@astrojs/starlight/components';
import { ShowDecisionCard } from '@noodlestan/designer-shows/astro';

## Props

#### `d: string | DecisionInput`

Carries the `DecisionInput` data object, or identifies the decision by its `name` or `id`.

<Code code={`<ShowDecisionCard d="Green Foreground" />`} title={'page.mdx'} lang="astro" />

<CardGrid>
    <ShowDecisionCard d="Green Foreground" />
</CardGrid>

#### `value: boolean | object`

Provide this flag to display the decision value next to it.

<Code code={`<Color d="Green Foreground" value />`} title={'page.mdx'} lang="astro" />

<CardGrid>
    <ShowDecisionCard d="Green Foreground" value />
</CardGrid>

In the case of color decisions, you may want to display the value in
several color spaces at the same time.

<Code
    code={`<Color d="Green Foreground" value={['rgb', 'hsl']} />`}
    title={'page.mdx'}
    lang="astro"
/>

<CardGrid>
    <ShowDecisionCard d="Green Foreground" value={['rgb', 'hsl']} />
</CardGrid>

<CardGrid>
    <ShowDecisionCard d="Green Foreground" usage />
</CardGrid>

#### `meta: boolean | string`

Provide this flag to display all decision data next to it.

Equivalent to `<ShowDecisionCard d="..." model params description contexts usage />`

<Code
    code={`<ShowDecisionCard d="Green Foreground" value="hsl" meta />`}
    title={'page.mdx'}
    lang="astro"
/>

<CardGrid>
    <ShowDecisionCard d="Green Foreground" value="hsl" meta />
</CardGrid>

#### `noshow: boolean | string`

Provide this flag to hide the decision value visualization.

Combine `noshow` with `meta` to display only decision data.

<Code
    code={`<ShowDecisionCard d="Green Foreground" noshow value="rgb" meta />`}
    title={'page.mdx'}
    lang="astro"
/>

<CardGrid>
    <ShowDecisionCard d="Green Foreground" noshow value="rgb" meta />
</CardGrid>
